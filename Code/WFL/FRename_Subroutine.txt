SUBROUTINE FRENAME(STRING NUM,NAME,DIR);
BEGIN
%% READS IN TASK NUMBER AND DESIRED CSV FILE NAME
%%THEN CHANGES FILENAME
STRING DIR1,DIR2,TODAY,PK;
DIR1:="""MOVELISTS";
DIR2:="EXTRACT";
TODAY:=TIMEDATE(YYYYMMDD);
PK:="##REDACTED##""";

DIR:=DIR1 / DIR2 / TODAY / NUM ON PK;
CHANGE £DIR TO MOVELISTS/EXTRACT/£TODAY/£NAME;
DIR:=DIR1 / DIR2 / TODAY / NAME ON PK;
END FRENAME;
